-- Add password_set column to user_profiles
-- Tracks whether a portal (customer) user has set a permanent password
-- after their initial magic link sign-in.

ALTER TABLE user_profiles
ADD COLUMN IF NOT EXISTS password_set BOOLEAN NOT NULL DEFAULT false;

-- All existing customer users have not set a password yet
-- (non-customer users don't use this column, but false is a safe default)
COMMENT ON COLUMN user_profiles.password_set IS 'Whether the user has set a permanent password (used for portal customer accounts)';
