-- ============================================================
-- RLS Audit Hardening Migration
-- 2026-03-02
-- ============================================================
-- All DROP POLICY IF EXISTS for full idempotency (safe to re-run).
--
-- This migration addresses two categories of RLS issues:
--
-- 1. TABLES MISSING RLS ENTIRELY
--    Tables created via Supabase dashboard that were never given
--    ENABLE ROW LEVEL SECURITY or had only dashboard-created policies.
--
-- 2. WEAK "authenticated = full access" POLICIES
--    AR/AP, config, and audit tables currently use USING (true)
--    for all authenticated users. We tighten to role-based policies.
--
-- Pattern used:
--   (select get_user_role()) — cached once per query
--   (select auth.uid())      — cached once per query
--   Staff (read) = admin, dispatcher, accounting
--   Financial (write) = admin, accounting
-- ============================================================


-- ============================================================
-- SECTION 1: Enable RLS on tables that may have been created
-- via Supabase dashboard. Drop any existing policies first.
-- ============================================================

-- trucks (used by equipment page, managed by admin/dispatcher)
ALTER TABLE trucks ENABLE ROW LEVEL SECURITY;

DROP POLICY IF EXISTS "Staff read trucks" ON trucks;
DROP POLICY IF EXISTS "Admin insert trucks" ON trucks;
DROP POLICY IF EXISTS "Admin update trucks" ON trucks;
DROP POLICY IF EXISTS "Admin delete trucks" ON trucks;
-- Also drop common dashboard-generated policy names
DROP POLICY IF EXISTS "Enable read access for authenticated users" ON trucks;
DROP POLICY IF EXISTS "Enable insert for authenticated users only" ON trucks;
DROP POLICY IF EXISTS "Enable update for authenticated users only" ON trucks;
DROP POLICY IF EXISTS "Enable delete for authenticated users only" ON trucks;
DROP POLICY IF EXISTS "Staff read all trucks" ON trucks;
DROP POLICY IF EXISTS "Staff update trucks" ON trucks;
DROP POLICY IF EXISTS "Staff insert trucks" ON trucks;
DROP POLICY IF EXISTS "Staff delete trucks" ON trucks;

CREATE POLICY "Staff read trucks"
  ON trucks FOR SELECT TO authenticated
  USING ((select get_user_role()) IN ('admin', 'dispatcher', 'accounting'));

CREATE POLICY "Admin insert trucks"
  ON trucks FOR INSERT TO authenticated
  WITH CHECK ((select get_user_role()) IN ('admin', 'dispatcher'));

CREATE POLICY "Admin update trucks"
  ON trucks FOR UPDATE TO authenticated
  USING ((select get_user_role()) IN ('admin', 'dispatcher'))
  WITH CHECK ((select get_user_role()) IN ('admin', 'dispatcher'));

CREATE POLICY "Admin delete trucks"
  ON trucks FOR DELETE TO authenticated
  USING ((select get_user_role()) = 'admin');

DROP POLICY IF EXISTS "Admin full access on containers" ON containers;
DROP POLICY IF EXISTS "Dispatcher read containers" ON containers;

CREATE POLICY "Staff read containers"
  ON containers FOR SELECT TO authenticated
  USING ((select get_user_role()) IN ('admin', 'dispatcher', 'accounting'));

CREATE POLICY "Drivers read assigned containers"
  ON containers FOR SELECT TO authenticated
  USING (
    (select get_user_role()) = 'driver'
    AND id IN (
      SELECT container_id FROM loads
      WHERE driver_id = (select auth.uid())
        AND container_id IS NOT NULL
    )
  );

CREATE POLICY "Staff insert containers"
  ON containers FOR INSERT TO authenticated
  WITH CHECK ((select get_user_role()) IN ('admin', 'dispatcher'));

  WITH CHECK ((select get_user_role()) IN ('admin', 'dispatcher'));

  USING ((select get_user_role()) = 'admin');

-- warehouse_inventory (referenced in dashboard warehouse module)
-- Only create if table exists (may not be in all environments)
DO $$ BEGIN
  IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_name = 'warehouse_inventory' AND table_schema = 'public') THEN
    EXECUTE 'ALTER TABLE warehouse_inventory ENABLE ROW LEVEL SECURITY';
    EXECUTE 'DROP POLICY IF EXISTS "Staff read warehouse_inventory" ON warehouse_inventory';
    EXECUTE 'DROP POLICY IF EXISTS "Staff insert warehouse_inventory" ON warehouse_inventory';
    EXECUTE 'DROP POLICY IF EXISTS "Staff update warehouse_inventory" ON warehouse_inventory';
    EXECUTE 'DROP POLICY IF EXISTS "Admin delete warehouse_inventory" ON warehouse_inventory';
    EXECUTE 'DROP POLICY IF EXISTS "Staff manage warehouse_inventory" ON warehouse_inventory';
    EXECUTE 'DROP POLICY IF EXISTS "Enable read access for authenticated users" ON warehouse_inventory';

    EXECUTE $policy$
      CREATE POLICY "Staff read warehouse_inventory"
        ON warehouse_inventory FOR SELECT TO authenticated
        USING ((select get_user_role()) IN ('admin', 'dispatcher', 'accounting'))
    $policy$;

    EXECUTE $policy$
      CREATE POLICY "Staff insert warehouse_inventory"
        ON warehouse_inventory FOR INSERT TO authenticated
        WITH CHECK ((select get_user_role()) IN ('admin', 'dispatcher'))
    $policy$;

    EXECUTE $policy$
      CREATE POLICY "Staff update warehouse_inventory"
        ON warehouse_inventory FOR UPDATE TO authenticated
        USING ((select get_user_role()) IN ('admin', 'dispatcher'))
        WITH CHECK ((select get_user_role()) IN ('admin', 'dispatcher'))
    $policy$;

    EXECUTE $policy$
      CREATE POLICY "Admin delete warehouse_inventory"
        ON warehouse_inventory FOR DELETE TO authenticated
        USING ((select get_user_role()) = 'admin')
    $policy$;
  END IF;
END $$;
