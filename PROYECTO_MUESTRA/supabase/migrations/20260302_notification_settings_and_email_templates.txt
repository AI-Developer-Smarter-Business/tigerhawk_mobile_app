-- ─── 1. Notification Settings ────────────────────────────────
-- Stores system-wide notification preferences (singleton row per setting key).
-- Admin can toggle alert types, set thresholds, configure recipients.
CREATE TABLE IF NOT EXISTS notification_settings (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  setting_key TEXT NOT NULL UNIQUE,         -- e.g. 'demurrage_alerts', 'unassigned_alerts', 'vessel_alerts', 'settlement_notifications'
  enabled BOOLEAN NOT NULL DEFAULT true,
  config JSONB NOT NULL DEFAULT '{}',       -- Flexible config per setting (thresholds, recipients, timing, etc.)
  updated_by UUID REFERENCES auth.users(id),
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- RLS
ALTER TABLE notification_settings ENABLE ROW LEVEL SECURITY;

-- All staff can read notification settings
DROP POLICY IF EXISTS "Staff read notification_settings" ON notification_settings;
CREATE POLICY "Staff read notification_settings"
  ON notification_settings FOR SELECT TO authenticated
  USING (
    (select get_user_role()) IN ('admin', 'dispatcher', 'accounting')
  );

-- Only admin can modify
DROP POLICY IF EXISTS "Admin manage notification_settings" ON notification_settings;
CREATE POLICY "Admin manage notification_settings"
  ON notification_settings FOR ALL TO authenticated
  USING ((select get_user_role()) = 'admin')
  WITH CHECK ((select get_user_role()) = 'admin');

-- ─── 2. Email Templates ─────────────────────────────────────
-- Editable email templates for portal invites, notifications, etc.
-- Supabase Auth emails (magic link, confirm) are configured in Supabase dashboard,
-- but this table lets admins manage custom notification templates.
CREATE TABLE IF NOT EXISTS email_templates (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  template_key TEXT NOT NULL UNIQUE,          -- e.g. 'portal_invite', 'settlement_ready', 'load_status_update'
  name TEXT NOT NULL,                         -- Human-friendly name
  subject TEXT NOT NULL,                      -- Email subject line (supports {{variables}})
  body_html TEXT NOT NULL,                    -- HTML body (supports {{variables}})
  variables TEXT[] NOT NULL DEFAULT '{}',     -- Available variables: ['customer_name', 'load_number', etc.]
  is_active BOOLEAN NOT NULL DEFAULT true,
  updated_by UUID REFERENCES auth.users(id),
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- RLS
ALTER TABLE email_templates ENABLE ROW LEVEL SECURITY;

-- All staff can read email templates
DROP POLICY IF EXISTS "Staff read email_templates" ON email_templates;
CREATE POLICY "Staff read email_templates"
  ON email_templates FOR SELECT TO authenticated
  USING (
    (select get_user_role()) IN ('admin', 'dispatcher', 'accounting')
  );

-- Only admin can modify
DROP POLICY IF EXISTS "Admin manage email_templates" ON email_templates;
CREATE POLICY "Admin manage email_templates"
  ON email_templates FOR ALL TO authenticated
  USING ((select get_user_role()) = 'admin')
  WITH CHECK ((select get_user_role()) = 'admin');
